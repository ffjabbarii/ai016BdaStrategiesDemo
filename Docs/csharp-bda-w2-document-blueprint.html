[HttpPost("blueprint/project/{projectName}/upload")]
public async Task<IActionResult> UploadDocumentToProject(
    string projectName, 
    [FromForm] IFormFile file)
{
    var result = await _blueprintProcessor
        .UploadDocumentToProjectAsync(projectName, fileBytes, file.FileName);
    
    return Ok(new {
        Status = "success",
        ProjectName = projectName,
        Language = "C#",
        Result = result
    });
}
